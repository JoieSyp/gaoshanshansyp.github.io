this的使用总结

全局作用域直接调用 指向window

var haha = “哈哈”；
console.log(this.haha);
//指向window

普通函数调用

function fn9 () {
    this.name = '123';
    this.age = '30';
}
window.fn9();
//指向window

function fn9 () {
    function heihei(){
    console.log(this+'heihei')
    }
    console.log(this+''123);
}
window.fn9();
//指向window

作为对象属性调用 指向该对象

varv obj = {
    name:'123'
    fn3 :function(){
        console.log(this.name);    
    }
}
obj.fn3();
//指向该对象

call apply的使用

var age = 20;

var obj = {
    age:"18",
    fn2:function(){
        console.log(this.age)
    }
}
obj.fn2.call(this,window);
